minimum(A){
    min = A[0]
    for i = 1 to A.length {
        if min > A[i] {
            min = A[i]
        }
    }
    return min
}

print(minimum([0, 1, 2]))

listInsert(L, x){
    x.next = L.head
    if L.head != 0 {
        L.head.prev = x
    }
    L.head = x
    x.prev = 0
    return x
}
